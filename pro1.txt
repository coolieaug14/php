<form action="" method="get">
    1. Armstrong number<br>
    2. Sum of digits<br>
    3. Display triangle<br>
    Enter your choice: <input type="number" name="choice"/><br>
    Enter a number: <input type="number" name="num"/><br>
    <input type="submit" value="Submit"/>
</form>

<?php
if (isset($_GET['choice'])) {
    $choice = (int) $_GET['choice']; // type casting
    $num = isset($_GET['num']) ? (int) $_GET['num'] : 0;

    switch ($choice) {
        case 1: // Armstrong number
            echo "<b><i>ARMSTRONG NUMBER</i></b><br>";
            $total = 0; 
            $x = $num;
            while ($num > 0) {
                $rem = $num % 10;
                $total += $rem ** 3;
                $num = (int)($num / 10);
            }
            echo ($x == $total) ? "$x is an Armstrong number" : "$x is not an Armstrong number";
            break;

        case 2: // Sum of digits
            echo "<b><i>SUM OF DIGITS</i></b><br>";
            $sum = 0;
            $temp = $num;
            while ($temp > 0) {
                $sum += $temp % 10;
                $temp = (int)($temp / 10);
            }
            echo "Sum of digits of $num = $sum";
            break;

        case 3: // Triangle
            echo "<b><i>TRIANGLE</i></b><br>";
            for ($i = 0; $i <= 5; $i++) {
                echo str_repeat("*", 5 - $i) . "<br>";
            }
            break;

        default:
            echo "Invalid choice!";
    }
}
?>
